469  void setupPinMode() {
470  AD_LOGD("DriverPins::setupPinMode");
471  // setup pins
472  for (auto &tmp : pins) {
473  if (tmp.pin != -1) {
474  if (!hasConflict(tmp.pin)) {
475  AD_LOGD("pinMode for %d", tmp.pin);
476  switch (tmp.pin_logic) {
477  case PinLogic::InputActiveHigh:
478  pinMode(tmp.pin, INPUT);
479  break;
480  case PinLogic::InputActiveLow:
481  pinMode(tmp.pin, INPUT_PULLUP);
482  break;
483  case PinLogic::Input:
484  pinMode(tmp.pin, INPUT);
485  break;
486  case PinLogic::Output:
487  pinMode(tmp.pin, OUTPUT);
488  break;
489  default:
490  // do nothing
491  break;
492  }
493  } else {
494  AD_LOGW("Pin '%d' not set up because of conflict", tmp.pin);
495  tmp.active = false;
496  }
497  } else {
498  AD_LOGD("Pin is -1");
499  }
500  }
501  }
